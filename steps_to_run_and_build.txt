# 1. Build container (one time)
docker build --platform=linux/amd64 -t adobe-hackathon .

# 2. Run Round 1A (PDF processing)
docker run --rm --platform=linux/amd64 -v "${PWD}/app/input:/app/input" -v "${PWD}/app/output:/app/output" -e ROUND=1A adobe-hackathon

# 3. Setup collections (prepare for Round 1B)
docker run --rm --platform=linux/amd64 -v "${PWD}/app/output:/app/output" -v "${PWD}/collections:/app/collections" adobe-hackathon python scripts/setup_collections.py

# 4. Run Round 1B (persona analysis)
docker run --rm --platform=linux/amd64 -v "${PWD}/collections:/app/collections" -e ROUND=1B adobe-hackathon
